
CREATE TABLE IF NOT EXISTS `adhi_user_profile_progress` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` int(11) NOT NULL,
  `state_exam_applied` tinyint(1) NOT NULL COMMENT 'whether user attended state exam',
  `obtained_license` tinyint(1) NOT NULL COMMENT 'whether user obtained license',
  `created_at` datetime NOT NULL,
  `updated_at` datetime NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1 AUTO_INCREMENT=1 ;
